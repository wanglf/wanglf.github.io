Freeradius是一款开源的AAA认证服务器, 本文总结Freeradius如何通过NTLM_AUTH接口将微软AD用户作为AAA认证用户数据库。本案例在CentOS 7环境下测试通过。

安装服务

#+BEGIN_EXMAPLE yum -y install samba samba-client samba-winbind krb5-server #+END_EXAMPLE

配置samba与winbind

/etc/samba/smb.conf

[global]
    workgroup = OA
    security = ads
    winbind use default domain = no
    passdb backend = tdbsam
    password server = 1.1.1.1
    realm = OA.ABC.COM

/etc/krb5.conf

[logging]
  default = FILE:/var/log/krb5libs.log
  kdc = FILE:/var/log/krb5kdc.log
  admin_server = FILE:/var/log/kadmind.log

[libdefaults]
  dns_lookup_realm = false
  ticket_lifetime = 24h
  renew_lifetime = 7d
  forwardable = true
  rdns = false
  default_realm = oa.abc.com
  #default_ccache_name = KEYRING:persistent:%{uid}

[realms]
  oa.abc.com = {
    kdc = 1.1.1.1:88
    admin_server = 1.1.1.1:749
  }

[domain_realm]
  .oa.abc.com = oa.abc.com
  oa.abc.com = oa.abc.com

[kdc]
 profile = /var/kerberos/krb5kdc/kdc.conf
[appdefaults]
  pam = {
    debug = false
    ticket_lifetime = 36000
    renew_lifetime = 36000
    forwardable = true
    krb4_convert = false
  }

/etc/nsswitch.conf

passwd:     files sss winbind
shadow:     files sss winbind
group:      files sss winbind
protocols:  files winbind
services:   files sss winbind
netgroup:   files sss winbind
automount:  files winbind

启用服务

#+BEGIN_SRC sh systemctl enable smb systemctl enable winbind systemctl start smb systemctl start winbind #+END_SRC sh

测试

ntlm_auth --request-nt-key --domain=OA --username=wesley

正常情况返回NT_STATUS_OK : Success (0x0)

Freeradius配置

/etc/raddb/mods-enabled/ntlm_auth

#+BEGIN_EXAMPLE exec ntlm_auth { wait = yes program = "/usr/bin/ntlm_auth –request-nt-key –domain=OA.ABC.COM –username=%{mschap:User-Name} –password=%{User-Password}" } #+END_SRC
